## Strike Craft as Fleet Support

# There is only one fighter class, which have all tweaks specific to it there

}


## Bombers - multipurpose fleet support role
#
# - Good against all ships, best against none
# - Can chase and hit corvettes easily
# - Ignore shields completely (unlike autocannon corvettes)
# - Good at finishing ships off, due to hull damage, no bonus vs armor
# - Not as good against heavy ships as heavy weapons

strike_craft_component_template = {
	key = "STRIKE_CRAFT_HANGAR_1"
	size = LARGE
	entity = "bomber_entity"
	power = @power1
	cost = @cost1

	count = @count
	regeneration_per_day = @regen
	launch_time = @launch_time

	damage = { min = 2 max = 8 }
	cooldown = @attack_cooldown
	range = @attack_range
	accuracy = 1.0
	tracking = @tracking1

	health = @health1
	armor = 0
	shield = @shield1
	evasion = @evasion1

	speed = @speed
	rotation_speed = @rotation
	acceleration = @acceleration

	shield_penetration = @attack_shield_pen
	armor_damage = @attack_armor
	hull_damage = @attack_hull

	ship_behavior = "scafs_bombers_behavior"

	weapon_type = instant
	projectile_gfx = "flak_pd_1"

	icon = "GFX_ship_part_strike_craft_bomber_1"
